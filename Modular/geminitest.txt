(async () => {
  // This is a sample transaction.
  const testTransaction = {
    description: "COFFEE SHOP PURCHASE ANYTOWN",
    amount: -4.50
  };

  try {
    console.log("Sending test transaction to Gemini...");
    // Access the geminiService through your global app object and call the function.
    const result = await window.financeApp.geminiService.categorizeTransaction(testTransaction, { skipAI: false });

      console.warn("The API call worked, but a category wasn't returned. Check the response object.");
    }

  } catch (error) {
    console.error("❌ Gemini API call failed. Check your API key, network connection, and the error details below.", error);
  }
})();
